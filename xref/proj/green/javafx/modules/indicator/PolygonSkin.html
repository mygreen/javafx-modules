29      @Override
30      public Shape createBlade(final double opacity, final int index) {
31          
32          //TODO: bindする
33          
34          final Polygon blade = PolygonBuilder.create()
35                  .points(0.0d, 0.0d,
36                          control.getWeight()/4.0, control.getSize()*1.0,
37                          control.getWeight()/4.0*3, control.getSize()*1.0,
38                          control.getWeight()*1.0, 0.0d)
39                  .fill(control.getColor())
40                  .opacity(opacity)
41                  .build();
42          return blade;
43      }
44
