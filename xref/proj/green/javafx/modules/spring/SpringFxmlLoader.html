80      /**
81       * コントローラを伴うFXMLを読み込む。
82       * <p>FXMLファイルは、"<コントローラ名>.fxml"とする。
83       * <p>リソースファイルも<コントローラ名>.properties"とする。ただし、存在すれば読み込む。
84       * @param controllerClass
85       */
86      @SuppressWarnings({"unchecked", "rawtypes"})
87      public <N extends Node, C> NodeAndController<N, C> loadNodeAndControllerFromSpring(final Class<N> nodeClass, final Class<C> controllerClass, final Locale locale) {
88          
89          final FXMLLoader loader = new FXMLLoader();
90          
91          // リソースの設定
92          ResourceBundle resources = getControllerResources(controllerClass, locale);
93          if(resources != null) {
94              loader.setResources(resources);
95          }
96          
97          /*
98           * ベースURLの設定
99           * 画像を読み込む際などに必要
100          */
101         final String basePath = "/".concat(controllerClass.getPackage().getName().concat("/")).replaceAll("\\.", "/");
102         loader.setLocation(getClass().getResource(basePath));
103         
104         // Spring用のControllerのCallbackの設定
105         loader.setControllerFactory(new Callback<Class<?>, Object>() {
106             @Override
107             public Object call(final Class<?> param) {
108                 
109                 // ControllerをSprign Beanコンテナから取得する
110                 final String beanName = getSpringBeanNameFromClassName(param);
111                 final Object beanObj = applicationContext.getBean(beanName);
112                 if(isPrototypeBean(beanName, beanObj)) {
113                     prototypeBeans.add(new PrototypeBean(beanName, controllerClass, beanObj));
114                 }
115                 return beanObj;                
116                 
117             }
118         });
119         
120         final String fxmlPath = "/".concat(controllerClass.getCanonicalName()).replaceAll("\\.", "/").replaceFirst("Controller", "").concat(".fxml");
121         final N node;
122         try {
123             node = (N) loader.load(getClass().getResourceAsStream(fxmlPath));
124             
125         } catch (IOException e) {
126             throw new RuntimeException(
127                     String.format("fail load fxml : %s", controllerClass.getClass().getCanonicalName()),
128                     e);
129         }
130         
131         // CSSの設定
132         if(node instanceof Parent) {
133             final URL stylesheet = getControllerCssUrl(controllerClass);
134             if(stylesheet != null) {
135                 ((Parent) node).getStylesheets().add(stylesheet.toExternalForm());
136             }
137         }
138         
139         // Controllerの取得
140         final C controller = loader.getController();
141         
142         return new NodeAndController(node, controller);
143         
144     }

161      * <p>通常、Controllerは<code><fx:root /></code>の属性「type」でしていたJavaFXのノードクラスを継承している。

169     public void loadComponentFromSpring(final Object root, final Object controller, final Locale locale) {

171         loadComponent(root, controller, locale);

174         injectSpringBean(controller);

177     /**
178      * Spring Bean名をクラス名から逆引きして取得する。
179      * <p>クラス名の先頭の文字を小文字にする。
180      * @param clazz
181      * @return
182      */
183     protected String getSpringBeanNameFromClassName(final Class<?> clazz) {
184         Assert.notNull(clazz);
185         String className = clazz.getSimpleName();
186         
187         className = String.valueOf(className.charAt(0)).toLowerCase().concat(className.substring(1));
188         return className;
189         
190     }

192     /**
193      * 任意のオブジェクトに対してSpring Beanをインジェクションする
194      * @param obj インジェクション対象のインスタンス
195      */
196     public void injectSpringBean(final Object obj) {
197         Assert.notNull(obj);
198         
199         SpringAnnotationInjector springInjector = (SpringAnnotationInjector) applicationContext.getBean(
200                 getSpringBeanNameFromClassName(SpringAnnotationInjector.class));
201         springInjector.inject(obj);
202         
203     }

205     /**
206      * SpringのBeanがSingletonかどうか判定する
207      * @param name
208      * @param beanObj
209      * @return
210      */
211     protected boolean isPrototypeBean(final String name, final Object beanObj) {
212         
213         if(applicationContext == null) {
214             logger.warning("applicationContext should be not null.");
215             return false;
216         }
217         
218         if(name == null || name.isEmpty()) {
219             logger.warning("bean name is empty.");
220             return false;
221         }
222         
223         if(!(applicationContext instanceof AbstractApplicationContext)) {
224             logger.warning("applicationContext is not AbstractApplicationContext");
225         }
226         
227         return ((AbstractApplicationContext) applicationContext).isPrototype(name);
228     }

230     /**
231      * 今まで生成した、Scope=Prototypeのオブジェクトを削除する。
232      * @param beanObjList SpringのBeanオブジェクト
233      */
234     
235     public void destroyAllProrotypeObject() {
236         
237         if(applicationContext == null) {
238             logger.warning("applicationContext should be not null.");
239             return;
240         }
241         
242         if(!(applicationContext instanceof AbstractApplicationContext)) {
243             if(logger.isLoggable(Level.INFO)) {
244                 logger.info("applicationContext is not AbstractApplicationContext");
245             }
246             
247             return;
248         }
249         
250         final ConfigurableBeanFactory beanFactory = 
251                 ((AbstractApplicationContext) applicationContext).getBeanFactory();
252         
253         for(PrototypeBean bean : getPrototypeBeans()) {
254             if(bean.isNotNullBeanInstance()) {
255                 beanFactory.destroyBean(bean.getName(), bean.getBeanInstance());
256                 
257             }
258         }
259         
260         getPrototypeBeans().clear();
261     }
