7   package proj.green.javafx.modules.constraint;
8   
9   import javafx.beans.property.ReadOnlyProperty;
10  import javafx.beans.value.ChangeListener;
11  import javafx.beans.value.ObservableValue;
12  import javafx.scene.control.TextInputControl;
13  
14  
15  /**
16   * Controlの最大文字数の制限をするリスナー。
17   * 
18   * @version 01-00
19   * @since 01-00
20   * @author T.TSUCHIE
21   *
22   */
23  public class MaxLengthListener implements ChangeListener<String> {
24      
25      /**
26       * 入力可能な最大文字数
27       */
28      final int maxLength;
29      
30      public MaxLengthListener(final int maxLength) {
31          
32          if(maxLength <= 0) {
33              throw new IllegalArgumentException("maxLength shoud be lather thane zero (> 0)");
34          }
35          
36          this.maxLength = maxLength;
37      }
38  
39      @Override
40      public void changed(final ObservableValue<? extends String> observable, final String oldValue, final String newValue) {
41          if(maxLength < newValue.length()) {
42              // 入力を受け付けない場合は、textプロパティを変更前に戻す。
43              final TextInputControl control = (TextInputControl)((ReadOnlyProperty)observable).getBean();
44              control.setText(oldValue);
45          }
46          
47      }
48      
49      
50  }
